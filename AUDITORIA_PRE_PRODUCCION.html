🎯 1.14 Middleware - Sin Manejo de Token Expirado

MEDIO Problema: El middleware valida token contra /api/v1/auth/me en cada request a rutas admin, sin cache.

📁 /frontend/middleware.js:14-22

✅ Solución - Cache de validación:
// Usar Next.js Edge Runtime con cache
export const config = {
    matcher: ['/admin/:path*'],
    runtime: 'edge'
};

// Cache de validación (5 minutos)
const tokenCache = new Map();

export async function middleware(request) {
    const token = request.cookies.get('access_token')?.value;

    // Verificar cache
    const cached = tokenCache.get(token);
    if (cached && Date.now() < cached.expires) {
        if (!cached.isAdmin) {
            return NextResponse.redirect(new URL('/', request.url));
        }
        return NextResponse.next();
    }

    // Validar con backend
    try {
        const res = await fetch(`${process.env.NEXT_PUBLIC_API_URL}/api/v1/auth/me`, {
            headers: { Authorization: `Bearer ${token}` },
            // Cache en edge
            next: { revalidate: 300 } // 5 minutos
        });

        if (!res.ok) throw new Error('Unauthorized');

        const user = await res.json();

        // Guardar en cache
        tokenCache.set(token, {
            isAdmin: user.rol === 'admin',
            expires: Date.now() + 300000 // 5 min
        });

        if (user.rol !== 'admin') {
            return NextResponse.redirect(new URL('/', request.url));
        }
    } catch (err) {
        return NextResponse.redirect(new URL('/login', request.url));
    }

    return NextResponse.next();
}

    response: Response,

    usuario: Optional[Usuario] = Depends(get_current_user_optional)
):
    if usuario:
        # Usuario autenticado: usar carrito de DB
        carrito = await carrito_service.obtener_carrito_usuario(usuario.id)

📧 2.5 Email Enumeration en Password Recovery

MEDIO Problema: El endpoint revela si un email existe en la base de datos.

📁 /backend/app/api/v1/auth.py:120-135

✅ Solución - Respuesta genérica:
@router.post("/recuperar-password")
@limiter.limit("3/hour")
async def recuperar_password(
    request: Request,
    data: RecuperarPasswordRequest,
    db: AsyncSession = Depends(get_db)
):
    # Siempre devolver el mismo mensaje (timing attack resistant)
    mensaje = (
        "Si el email existe en nuestra base de datos, "
        "recibirás un link de recuperación."
    )

    # Buscar usuario
    result = await db.execute(
        select(Usuario).where(Usuario.email == data.email)
    )
    usuario = result.scalar_one_or_none()

    if usuario:
        # Generar token
        token = secrets.token_urlsafe(32)
        expires = datetime.now(timezone.utc) + timedelta(hours=24)

        usuario.reset_token = token
        usuario.reset_token_expires = expires
        await db.commit()

        # Enviar email (en background para no revelar timing)
        background_tasks.add_task(
            enviar_email_recuperacion,
            usuario.email,
            token
        )
    else:
        # ⚠️ IMPORTANTE: Agregar delay artificial
        # para que tome el mismo tiempo que enviar email
        await asyncio.sleep(0.5)

    # Siempre devolver 200 OK con mismo mensaje
    return {"message": mensaje}
